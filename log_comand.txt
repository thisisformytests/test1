git reset --hard -откатывает все изменение и возращает все файлы в исходное состояние
git restore filename or . -возвращает удаленные файлы
git checkout -b имя -создает новую ветку   git branch имя
git checkout имя перемещение по веткам
git merge слияние

git push origin --delete имя -удалить ветку на сервере
